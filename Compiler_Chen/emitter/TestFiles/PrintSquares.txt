VAR 
count, ignore, times: int;
PROCEDURE printSquares(low, high:int);
VAR count, square: int;
BEGIN
count := low;
WHILE count <= high DO
BEGIN
square := count * count;
WRITELN(square);
count := count + 1;
times := times + 1;
END;
END;

BEGIN
count := 196;
times := 0;
printSquares(10, 13);
WRITELN(count);
WRITELN(times);
END;
